```
function solution(word) {
    var answer = 0;
    let vowels = ['A','E','I','O','U'];
    let list = [];
    
    /**
      * 차례대로 모음 사전을 모아서 word 파라미터에 대한 인덱스 획득 
      */
    const dfs = (cur, len) => {
        if(len == 6) {
            console.log(cur);
            return ;
        } 
        
        list.push(cur);
        
        for(let i=0; i<vowels.length; i++) {
            dfs(cur+vowels[i], len+1);
        }
    }
    
    dfs("",0); //A가 1부터 시작 
    
    return list.indexOf(word);
}
```
