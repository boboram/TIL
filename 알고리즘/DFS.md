# JS 코드 
```
// 하나씩 둘러보면서 방문하지 않은 인접 노드들을 다시 방문..
//가장 작은 1의 인접노드들부터 둘러보면서
//1의 인접노드들은 [2,3,8] 이니까 다시 2의 인접노드들 검색
//2의 인접노드는 [1,7]인데 1은 방문했으니 7의 인접노드들 검색
//7의 인접노드는 [2,6,8] 인데 2는 방문했으니 6의 인접노드들 검색
//6의 인접노드는 [7]인데 이미 방문했으니 넘어가고
//7의 인접노드에서 8을 방문하지 않았으니 8의 인접노드들 검색
//8의 인접노드는 [1,7]인데 모두 방문했으니 넘어가고
//2의 인접노드에서 1도 이미 방문했으니 넘어가고
//1의 인접노드에서 방문하지 않은 3의 인접노드들 검색
//3의 인접노드는 [1,4,5] 인데 1은 방문했고 4의 인접노드들 검색
//4의 인접노드는 [3,5]인데 3은 방문했고 5 방문 안했으니 4의 인접노드들 검색
//5의 인접노드는 [3,4]인데 모두 검색했으니 넘어가고
//다시 1의 인접노드에서 8은 이미 방문했으니 끝! 
function dfs(v, graph, visited) {
    console.log(v--); 
    visited[v] = true;
    const c_graph = graph[v];

    for(let i=0; i<c_graph.length; i++) {
        if(!visited[c_graph[i]-1]) { 
            dfs(c_graph[i], graph, visited);
        }
    }
}

const graph = [ [2,3,8], [1,7], [1,4,5], [3,5], [3,4], [7], [2,6,8], [1,7] ];
let visited = Array(graph.length).fill(false);
dfs(1, graph, visited);
```

# 결과 

```
1 -> 2 -> 7 -> 6 -> 8 -> 3 -> 4 -> 5
```
